let answer = 0;

  if (idx === numbers.length) {
    if (target === sum) {
      answer += 1;
    }
    return;
  } else {
    let tmp = sum + numbers[idx] * plusMinus;
    dfs(numbers, target, idx + 1, 1, tmp);
    dfs(numbers, target, idx + 1, -1, tmp);
  }
};

const solution = (numbers, target) => {
  dfs(numbers, target, 0, 1, 0);
  dfs(numbers, target, 0, -1, 0);
  return answer / 2;
};
